       });

    3. **Column rename — add onError rollback.** The current `renameColumn` already has `onMutate` that updates the cache, but lacks `onError` rollback. Add it:
       ```tsx
       const renameColumn = api.column.update.useMutation({
         onMutate: async ({ id, name, type }) => {
           await utils.column.getByTableId.cancel({ tableId });
           const previous = utils.column.getByTableId.getData({ tableId });
           utils.column.getByTableId.setData({ tableId }, (old) =>
             old?.map((col) =>
               col.id === id
                 ? { ...col, name, ...(type ? { type } : {}) }
                 : col,
             ),
           );
           return { previous };
         },
         onError: (_err, _vars, context) => {
           if (context?.previous !== undefined) {
             utils.column.getByTableId.setData({ tableId }, context.previous);
           }
           toast.error("Failed to rename column. Changes reverted.");
         },
         onSettled: () => {
           void utils.column.getByTableId.invalidate({ tableId });
         },
       });
       ```

    4. **Column delete — optimistic remove.** Replace the current `deleteColumn` mutation:
       ```tsx
       const deleteColumn = api.column.delete.useMutation({
         onMutate: async ({ id }) => {
           await utils.column.getByTableId.cancel({ tableId });
           const previous = utils.column.getByTableId.getData({ tableId });
           utils.column.getByTableId.setData({ tableId }, (old) =>
             old?.filter((col) => col.id !== id) ?? [],
           );
           // Also clear the page cache since cell data for this column is now invalid
           const prevPageCache = { ...pageCacheRef.current };
           // Remove the column's cells from all cached rows
           for (const [pageIdx, pageRows] of Object.entries(pageCacheRef.current)) {
             pageCacheRef.current[Number(pageIdx)] = pageRows.map((row) => {
               const { [id]: _removed, ...remainingCells } = row.cells;
               return { ...row, cells: remainingCells };
             });
           }
           forceUpdate();
           return { previous, prevPageCache };
         },
         onError: (_err, _vars, context) => {
           if (context?.previous !== undefined) {
             utils.column.getByTableId.setData({ tableId }, context.previous);
           }
           if (context?.prevPageCache) {
             pageCacheRef.current = context.prevPageCache;
             forceUpdate();
           }
           toast.error("Failed to delete column. Changes reverted.");
         },
         onSettled: () => {
           void utils.column.getByTableId.invalidate({ tableId });
         },
       });
       ```
       This replaces the old onSuccess-only pattern that cleared the page cache and re-fetched. Now the column is optimistically removed from the header and all cached rows, with full rollback on error.
