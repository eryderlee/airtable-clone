         - id: "seed-table-1"
         - name: "Million Row Table"
         - baseId: "seed-base-1"

       - Create 3 test columns:
         - { id: "col-name", tableId: "seed-table-1", name: "Name", type: "text", order: 0 }
         - { id: "col-age", tableId: "seed-table-1", name: "Age", type: "number", order: 1 }
         - { id: "col-email", tableId: "seed-table-1", name: "Email", type: "text", order: 2 }

       - Insert 1,000,000 rows in chunks of 1,000:
         ```typescript
         const TOTAL_ROWS = 1_000_000;
         const CHUNK_SIZE = 1_000;

         for (let i = 0; i < TOTAL_ROWS; i += CHUNK_SIZE) {
           const chunk = Array.from({ length: Math.min(CHUNK_SIZE, TOTAL_ROWS - i) }, (_, j) => ({
             tableId: "seed-table-1",
             rowOrder: i + j,
             cells: {
               "col-name": faker.person.fullName(),
               "col-age": faker.number.int({ min: 18, max: 90 }),
               "col-email": faker.internet.email(),
             },
           }));
           await db.insert(schema.rows).values(chunk);
           if ((i + CHUNK_SIZE) % 100_000 === 0) {
             console.log(`Inserted ${i + CHUNK_SIZE} / ${TOTAL_ROWS} rows`);
           }
         }
         ```

       - After all inserts, run ANALYZE:
         ```typescript
         import { sql } from "drizzle-orm";
         await db.execute(sql`ANALYZE "row"`);
         ```
         NOTE: Table name "row" must be quoted since it is a SQL reserved word.

       - Log total time and close the connection:

    1. Create `scripts/benchmark.ts`:

       - Connect via DATABASE_URL (transaction pool, port 6543) -- this simulates production app traffic.

       - Benchmark 1: First page query (no cursor):
         ```typescript
         const start = performance.now();
         const firstPage = await db
           .select()
           .from(schema.rows)
           .where(eq(schema.rows.tableId, "seed-table-1"))
           .orderBy(asc(schema.rows.rowOrder), asc(schema.rows.id))
           .limit(100);
         const firstPageTime = performance.now() - start;
         console.log(`First page (100 rows): ${firstPageTime.toFixed(2)}ms`);
         ```

       - Benchmark 2: Mid-table cursor query (cursor at row ~500,000):
         Use the last row from a query at offset ~500k to get a realistic cursor:
         ```typescript
         // Get a cursor near the middle
         const midRow = firstPage[firstPage.length - 1]!;
         // Simulate having paginated to middle by using a known rowOrder
         const midCursor = { rowOrder: 500_000, id: "zzz" }; // Use a real row's values

         const start2 = performance.now();
         const midPage = await db
           .select()
           .from(schema.rows)
           .where(
             and(
               eq(schema.rows.tableId, "seed-table-1"),
               or(
                 gt(schema.rows.rowOrder, midCursor.rowOrder),
                 and(
                   eq(schema.rows.rowOrder, midCursor.rowOrder),
                   gt(schema.rows.id, midCursor.id)
                 )
               )
             )
           )
           .orderBy(asc(schema.rows.rowOrder), asc(schema.rows.id))
           .limit(100);
         const midPageTime = performance.now() - start2;
         console.log(`Mid-table cursor page (100 rows): ${midPageTime.toFixed(2)}ms`);
         ```

       - Benchmark 3: End-of-table cursor query (cursor at row ~999,000):
         Same pattern with rowOrder: 999_000.

       - Benchmark 4: Larger page size (500 rows):
         Same cursor pattern with .limit(500).

         ```
         End-table cursor (100 rows): XXms
         ```
