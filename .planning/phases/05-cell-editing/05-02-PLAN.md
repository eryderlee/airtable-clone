    ```typescript
    const utils = api.useUtils(); // Should already exist from 04-02

    const updateCell = api.row.update.useMutation({
      onMutate: async ({ id, cells }) => {
        // Cancel outgoing refetches
        await utils.row.getRows.cancel({ tableId, viewId });

        // Snapshot previous data
        const previousData = utils.row.getRows.getInfiniteData({ tableId, viewId });

        // Optimistically update the cache
        utils.row.getRows.setInfiniteData({ tableId, viewId }, (old) => {
          if (!old) return old;
          return {
            ...old,
            pages: old.pages.map((page) => ({
              ...page,
              items: page.items.map((row) =>
                row.id === id ? { ...row, cells: { ...row.cells, ...cells } } : row
              ),
            })),
          };
        });

        return { previousData };
      },
      onError: (_err, _vars, context) => {
        // Rollback on error
        if (context?.previousData) {
          utils.row.getRows.setInfiniteData({ tableId, viewId }, context.previousData);
        }
      },
      onSettled: () => {
        // Refetch to sync with server
        void utils.row.getRows.invalidate({ tableId });
      },
    });
    ```
    CRITICAL: Use `{ tableId, viewId }` as the key for `setInfiniteData` and `cancel` -- this matches the query key shape used by `useInfiniteQuery`.

    ```typescript
    const handleKeyDown = useCallback((e: React.KeyboardEvent<HTMLDivElement>) => {
      if (!cursor) return;

      const { rowIndex, columnId } = cursor;
      const colIdx = columnOrder.indexOf(columnId);
      const isEditing = editingCell !== null;
      const totalRows = flatRows.length;

      // --- EDIT MODE ---
      if (isEditing) {
        // In edit mode, most keys are handled by the input itself.
        // Only intercept Tab (which the input doesn't handle)
        if (e.key === "Tab") {
          e.preventDefault();
          // Commit happens via onBlur when focus leaves the input
          // Then move cursor
          setEditingCell(null);
          setInitialDraft(undefined);

          if (e.shiftKey) {
            // Shift+Tab: move left
            if (colIdx > 0) {
              const prevColId = columnOrder[colIdx - 1]!;
              moveCursor(rowIndex, prevColId);
            } else if (rowIndex > 0) {
              // Wrap to end of previous row
              const lastColId = columnOrder[columnOrder.length - 1]!;
              moveCursor(rowIndex - 1, lastColId);
            }
          } else {
            // Tab: move right
            if (colIdx < columnOrder.length - 1) {
              const nextColId = columnOrder[colIdx + 1]!;
              moveCursor(rowIndex, nextColId);
            } else if (rowIndex < totalRows - 1) {
              // Wrap to start of next row
              const firstColId = columnOrder[0]!;
              moveCursor(rowIndex + 1, firstColId);
            }
          }
        }
        // Enter and Escape are handled by GridCell's input onKeyDown with stopPropagation
        return;
      }

      // --- NAVIGATION MODE ---
      switch (e.key) {
        case "ArrowUp":
          e.preventDefault();
          if (rowIndex > 0) {
            moveCursor(rowIndex - 1, columnId);
          }
          break;

        case "ArrowDown":
          e.preventDefault();
          if (rowIndex < totalRows - 1) {
            moveCursor(rowIndex + 1, columnId);
          }
          break;

        case "ArrowLeft":
          e.preventDefault();
          if (colIdx > 0) {
            const prevColId = columnOrder[colIdx - 1]!;
            moveCursor(rowIndex, prevColId);
          }
          break;

        case "ArrowRight":
          e.preventDefault();
          if (colIdx < columnOrder.length - 1) {
            const nextColId = columnOrder[colIdx + 1]!;
            moveCursor(rowIndex, nextColId);
          }
          break;

        case "Enter":
          e.preventDefault();
          setEditingCell({ rowIndex, columnId });
          setInitialDraft(undefined); // No initial draft -- select all text
          break;

        case "Escape":
          e.preventDefault();
          setCursor(null); // Deselect
          break;

        case "Tab":
          e.preventDefault();
          if (e.shiftKey) {
            if (colIdx > 0) {
              const prevColId = columnOrder[colIdx - 1]!;
              moveCursor(rowIndex, prevColId);
            } else if (rowIndex > 0) {
              const lastColId = columnOrder[columnOrder.length - 1]!;
              moveCursor(rowIndex - 1, lastColId);
            }
          } else {
            if (colIdx < columnOrder.length - 1) {
              const nextColId = columnOrder[colIdx + 1]!;
              moveCursor(rowIndex, nextColId);
            } else if (rowIndex < totalRows - 1) {
              const firstColId = columnOrder[0]!;
              moveCursor(rowIndex + 1, firstColId);
            }
          }
          break;

        default:
          // Printable character: enter edit mode with this character
          if (e.key.length === 1 && !e.ctrlKey && !e.metaKey && !e.altKey) {
            e.preventDefault();
            setEditingCell({ rowIndex, columnId });
            setInitialDraft(e.key);
          }
          break;
      }
    }, [cursor, editingCell, columnOrder, flatRows.length, moveCursor]);
    ```

    CRITICAL details:
    - `e.preventDefault()` on ALL navigation keys to prevent browser defaults (Tab focus, Arrow scroll, etc.)
    - Printable char detection: `e.key.length === 1` catches letters, numbers, symbols; excludes special keys (Shift, Control, etc.)
    - `!e.ctrlKey && !e.metaKey && !e.altKey` prevents intercepting Ctrl+C, Cmd+A, etc.
    - Tab wraps from end of row to start of next row (and vice versa with Shift+Tab)
    - Arrow keys do NOT wrap -- they stop at grid boundaries
    - In edit mode, only Tab is intercepted; Enter and Escape are handled by GridCell input's stopPropagation
